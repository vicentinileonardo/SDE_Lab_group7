In this page you can send a request both to GraphQL and REST.

In DEV TOOLS (network tab) you can see how requests are made.

        
async function movies(page = null) {
    console.log((await (fetch('http://localhost:3000/movies').then(r => r.json()))));
    console.log((await (fetch('http://localhost:4000/', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
            operationName: page === null ? 'highlighted' : 'movies',
            query: `
                query highlighted{
                    highlightedMovies {
                        title
                        genres
                    }
                }
                query movies($page: Int!) {
                    movies(page: $page) {
                        title
                        genres
                    }
                }
            `,
            variables: page === null ? { } : { page }
        }),
    }).then(r => r.json()))));
}
